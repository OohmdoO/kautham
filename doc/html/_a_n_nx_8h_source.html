112 //----------------------------------------------------------------------
113 // Orthogonal (axis aligned) rectangle
114 // Orthogonal rectangles are represented by two points, one
115 // for the lower left corner (min coordinates) and the other
116 // for the upper right corner (max coordinates).
117 //
118 // The constructor initializes from either a pair of coordinates,
119 // pair of points, or another rectangle. Note that all constructors
120 // allocate new point storage. The destructor deallocates this
121 // storage.
122 //
123 // BEWARE: Orthogonal rectangles should be passed ONLY BY REFERENCE.
124 // (C++'s default copy constructor will not allocate new point
125 // storage, then on return the destructor free's storage, and then
126 // you get into big trouble in the calling procedure.)
127 //----------------------------------------------------------------------
128 
129 class ANNorthRect {
130 public:
131  ANNpoint lo; // rectangle lower bounds
132  ANNpoint hi; // rectangle upper bounds
133 //
134  ANNorthRect( // basic constructor
135  int dd, // dimension of space
136  ANNcoord l=0, // default is empty
137  ANNcoord h=0)
138  { lo = annAllocPt(dd, l); hi = annAllocPt(dd, h); }
139 
140  ANNorthRect( // (almost a) copy constructor
141  int dd, // dimension
142  const ANNorthRect &r) // rectangle to copy
143  { lo = annCopyPt(dd, r.lo); hi = annCopyPt(dd, r.hi); }
144 
145  ANNorthRect( // construct from points
146  int dd, // dimension
147  ANNpoint l, // low point
148  ANNpoint h) // hight point
149  { lo = annCopyPt(dd, l); hi = annCopyPt(dd, h); }
150 
151  ~ANNorthRect() // destructor
152  { delete lo; delete hi; lo = hi = NULL; }
153 
154  ANNbool inside(int dim, ANNpoint p);// is point p inside rectangle?
155 };

162 //----------------------------------------------------------------------
163 // Orthogonal (axis aligned) halfspace
164 // An orthogonal halfspace is represented by an integer cutting
165 // dimension cd, coordinate cutting value, cv, and side, sd, which
166 // is either +1 or -1. Our convention is that point q lies in the
167 // (closed) halfspace if (q[cd] - cv)*sd >= 0.
168 //----------------------------------------------------------------------
169 
170 class ANNorthHalfSpace {
171 public:
172  int cd; // cutting dimension
173  ANNcoord cv; // cutting value
174  int sd; // which side
175 //
176  ANNorthHalfSpace() // default constructor
177  { cd = 0; cv = 0; sd = 0; }
178 
179  ANNorthHalfSpace( // basic constructor
180  int cdd, // dimension of space
181  ANNcoord cvv, // cutting value
182  int sdd) // side
183  { cd = cdd; cv = cvv; sd = sdd; }
184 
185  ANNbool in(ANNpoint q) const // is q inside halfspace?
186  { return (ANNbool) ((q[cd] - cv)*sd >= 0); }
187 
188  ANNbool out(ANNpoint q) const // is q outside halfspace?
189  { return (ANNbool) ((q[cd] - cv)*sd < 0); }
190 
191  ANNdist dist(ANNpoint q) const // (squared) distance from q
192  { return (ANNdist) ANN_POW(q[cd] - cv); }
193 
194  void setLowerBound(int d, ANNpoint p)// set to lower bound at p[i]
195  { cd = d; cv = p[d]; sd = +1; }
196 
197  void setUpperBound(int d, ANNpoint p)// set to upper bound at p[i]
198  { cd = d; cv = p[d]; sd = -1; }
199 
200  void project(ANNpoint &q) // project q (modified) onto halfspace
201  { if (out(q)) q[cd] = cv; }
202 };
