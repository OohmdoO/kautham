116  void tree_edge(Edge e, Graph& g)
117  {
118  typename graph_traits<Graph>::vertex_descriptor s=source(e,g);
119  typename graph_traits<Graph>::vertex_descriptor t=target(e,g);
120  //expand the wave
121  d[t] = d[s] + 1;
122  //control the expansion value by the distance to the obstacles
123  KthReal v = (KthReal)loc[t].d/maxdwall;
124  if(v>0.9) v=0.9;

128  template <typename Edge, typename Graph>
129  void gray_target(Edge e, Graph& g)
130  {
131  //for the vertices already visited, update their value if necessary
132  typename graph_traits<Graph>::vertex_descriptor s=source(e,g);
133  typename graph_traits<Graph>::vertex_descriptor t=target(e,g);
134  KthReal v = (KthReal)loc[t].d/maxdwall;
135  if(v>0.9) v=0.9;
136  KthReal newvalue = d[s] + 1 - v;
137  if(d[t] > newvalue)
138  d[t] = newvalue;
