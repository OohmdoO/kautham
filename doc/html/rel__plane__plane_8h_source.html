110 inline RelPlanePlane::RelPlanePlane(const Plane3& P1,
111  const Plane3& P2) :
112 
113  RelBaseAng(angle(P1.getNormal(),
114  P2.getNormal()))
115 {
116  // Angle type
117  m_type = angleType(m_angle);
118 
119  // Distance between planes
120  if (m_type.test(SAME_DIRECTION))
121  {
122  // Parallel/antiparallel planes
123  m_dist = distance(P1.getSupport(),
124  P2.getSupport());
125  }
126  else
127  {
128  // Planes intersect
129  m_dist = 0.0;
130  intersection(P1, P2);
131  }
132 
133  // Distance type
134  m_type |= distanceType(m_dist);
135 }

151 inline void RelPlanePlane::intersection(const Plane3& P1,
152  const Plane3& P2)
153 {
154  // Plane parameters
155  const Unit3 norm1(P1.getNormal());
156  const Scalar dist1(P1.getDistOrig());
157 
158  const Unit3 norm2(P2.getNormal());
159  const Scalar dist2(P2.getDistOrig());
160 
161  // Intersection line direction vector
162  const Unit3 int_dir = cross(norm1, norm2);
163 
164  // Intersection line support point
165  const Matrix3x3 A ( norm1[0], norm1[1], norm1[2],
166  norm2[0], norm2[1], norm2[2],
167  int_dir[0], int_dir[1], int_dir[2]);
168 
169  const Vector3 b(-dist1,
170  -dist2,
171  Scalar(0.0));
172 
173  const Point3 int_sup(inverse(A) * b);
174 
175  // Intersection line
176  m_intersection.setValue(int_dir,
177  int_sup,
178  UNDIRECTED);
179 }

183 {
