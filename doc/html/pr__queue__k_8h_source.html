85 //----------------------------------------------------------------------
86 // ANNmin_k
87 // An ANNmin_k structure is one which maintains the smallest
88 // k values (of type PQKkey) and associated information (of type
89 // PQKinfo). The special info and key values PQ_NULL_INFO and
90 // PQ_NULL_KEY means that thise entry is empty.
91 //
92 // It is currently implemented using an array with k items.
93 // Items are stored in increasing sorted order, and insertions
94 // are made through standard insertion sort. (This is quite
95 // inefficient, but current applications call for small values
96 // of k and relatively few insertions.)
97 //----------------------------------------------------------------------
98 
99 class ANNmin_k {
100  struct mk_node { // node in min_k structure
101  PQKkey key; // key value
102  PQKinfo info; // info field (user defined)
103  };
104 
105  int k; // max number of keys to store
106  int n; // number of keys currently active
107  mk_node *mk; // the list itself
108 
109 public:
110  ANNmin_k(int max) // constructor (given max size)
111  {
112  n = 0; // initially no items
113  k = max; // maximum number of items
114  mk = new mk_node[max+1]; // sorted array of keys
115  }
116 
117  ~ANNmin_k() // destructor
118  { delete [] mk; }
119 
120  PQKkey ANNmin_key() // return minimum key
121  { return (n > 0 ? mk[0].key : PQ_NULL_KEY); }
122 
123  PQKkey max_key() // return maximum key
124  { return (n == k ? mk[k-1].key : PQ_NULL_KEY); }
125 
126  PQKkey ith_smallest_key(int i) // ith smallest key (i in [0..n-1])
127  { return (i < n ? mk[i].key : PQ_NULL_KEY); }
128 
129  PQKinfo ith_smallest_info(int i) // info for ith smallest (i in [0..n-1])
130  { return (i < n ? mk[i].info : PQ_NULL_INFO); }
131 
132  inline void insert( // insert item (inlined for speed)
133  PQKkey kv, // key value
134  PQKinfo inf) // item info
135  {
136  register int i;
137  // slide larger values up
138  for (i = n; i > 0; i--) {
139  if (mk[i-1].key > kv)
140  mk[i] = mk[i-1];
141  else
142  break;
143  }
144  mk[i].key = kv; // store element here
145  mk[i].info = inf;
146  if (n < k) n++; // increment number of items
147  //FLOP(k-i+1) // increment floating ops
148  }
149 };
