114 inline RelLineCircle::RelLineCircle(const Line3& L,
115  const Circle3& c)
116 {
117  // Relation between line and circle support plane
118  const Plane3 sup = c.getSupportPlane();
119  const RelLinePlane rel(L, sup);
120  const RelType rel_type = rel.getType();
121 
122  // Circle parameters
123  const Point3 center = c.getCenter();
124  const Scalar radius = c.getRadius();
125  const Unit3 normal = c.getNormal();
126 
127  // Line direction
128  const Unit3 direction = L.getDirection();
129 
130  // Computes closest line point to circle center and projects it on the
131  // circle support plane
132  const Point3 proj_l = project(center, L);
133  const Point3 proj_p = project(proj_l, sup);
134 
135  // Distance between elements
136  if (rel_type.test(INTERSECTING) &&
137  proj_p == center)
138  {
139  // Line passes through circle center
140  m_type |= (rel_type.test(PERPENDICULAR)) ? CONCENTRIC : DISTANCE;
141  }
142  else if (rel_type.test(CONTAINED) &&
143  proj_p == center)
144  {
145  // Line is contained in circle plane and passes through circle center
146  const Vector3 offset = radius * direction;
147  m_p1 = center + offset;
148  m_p2 = center - offset;
149  m_type |= INTERSECTING;
150  }
151  else
152  {
153  // Closest circle and line points.
154  const Unit3 dir_c = proj_p - center;
155  const Point3 closest_c = center + radius * dir_c;
156  const Point3 closest_l = project(closest_c, L);
157  const Scalar dist_closest = distance(closest_l, closest_c);
158  const Scalar dist_center = distance(closest_l, center);
159 
160  if (dist_closest == Scalar(0.0))
161  {
162  // Tangent elements
163  m_p1 = closest_l;
164  m_type |= TANGENT;
165  }
166  else if(dist_center < radius &&
167  rel_type.test(CONTAINED))
168  {
169  // Intersecting elements
170  const Scalar d = sqrt(sq(radius) - distance(closest_l, center));
171  const Vector3 offset = d * direction;
172 
173  m_p1 = closest_l + offset;
174  m_p2 = closest_l - offset;
175  m_type |= INTERSECTING;
176  }
177  else if (dist_center < radius &&
178  rel_type.test(DISTANCE))
179  {
180  // Line is over circle (i.e., casts a shadow on it)
181  m_type |= DISTANCE;
182  }
183  else
184  {
185  m_type |= DISTANCE;
186  }
187  }
188 }

202 inline Point3 RelLineCircle::getIntersectionPoint(size_t n) const
203 {
204  util::Assert(m_type.test(INTERSECTING),
205  Exception("Line and cylinder do not intersect."));
